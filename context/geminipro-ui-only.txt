    <script>
        // Simple view router
        function switchView(viewId) {
            // Update Nav Links
            document.querySelectorAll('.nav-link').forEach(link => link.classList.remove('active'));
            event.target.classList.add('active');

            // Hide all views
            document.querySelectorAll('.app-view').forEach(view => {
                view.classList.remove('active');
            });

            // Show target view
            document.getElementById(viewId).classList.add('active');
        }

        // Mock Die Roll & Damage Demo (To show intentional motion/feedback)
        function mockRoll() {
            const die = document.getElementById('physical-die');
            const span = document.getElementById('die-result');
            
            if(die.classList.contains('rolling')) return;
            
            span.innerText = "?";
            die.classList.add('rolling');
            
            setTimeout(() => {
                die.classList.remove('rolling');
                const roll = Math.floor(Math.random() * 20) + 1;
                span.innerText = roll;
                
                // If it's a bad roll, demonstrate state changing (Ironmole takes damage)
                if(roll < 10) {
                    const hpBar = document.getElementById('ironmole-hp-bar');
                    const hpText = document.getElementById('ironmole-hp-text');
                    
                    hpBar.style.width = "40%";
                    hpBar.style.background = "var(--color-blood-700)";
                    hpBar.style.boxShadow = "0 0 10px var(--color-blood-700)";
                    hpText.innerText = "8 / 20";
                    hpText.style.color = "var(--color-blood-700)";
                    
                    // Flash the card border
                    const card = document.querySelector('.hero-card.ironmole');
                    card.style.borderColor = "var(--color-blood-700)";
                    setTimeout(() => {
                        card.style.borderColor = "var(--color-abyss-500)";
                    }, 500);
                }
            }, 800);
        }
    </script>
